🏥 CODE REFERENCE

Python Implementation

Complete pipeline from data loading to model evaluation

import pandas as pd
import numpy as np
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split

# Load & remove duplicates
medical = pd.read_csv('insurance.csv').drop_duplicates()

# Label encode categorical columns
medical['sex'] = LabelEncoder().fit_transform(medical['sex'])
medical['smoker'] = LabelEncoder().fit_transform(medical['smoker'])

# Log10 transform target to reduce right skew
medical['charges'] = np.log10(medical['charges'])

# Feature selection & 80/20 split
X = medical[['age', 'sex', 'bmi', 'smoker']]
Y = medical['charges']
X_train, X_test, Y_train, Y_test = train_test_split(
X, Y, test_size=0.2, random_state=42)
from sklearn.linear_model import LinearRegression
from sklearn.tree import DecisionTreeRegressor
from sklearn.ensemble import GradientBoostingRegressor

# 1. Linear Regression (Baseline)
lr = LinearRegression()
lr.fit(X_train, Y_train)

# 2. Decision Tree (with hyperparameters)
dt = DecisionTreeRegressor(
max_depth=30, min_samples_split=50,
min_samples_leaf=10, random_state=42
)
dt.fit(X_train, Y_train)

# 3. Gradient Boost (Best — no tuning needed!)
gb = GradientBoostingRegressor(random_state=42)
gb.fit(X_train, Y_train)
from sklearn.metrics import r2_score, mean_squared_error

for name, model in {'Linear Reg': lr, 'Decision Tree': dt, 'Gradient Boost': gb}.items():
pred = model.predict(X_test)
print(f"{name}")
print(f" Test R²: {r2_score(Y_test, pred):.4f}")
print(f" Test MSE: {mean_squared_error(Y_test, pred):.4f}")

# Output:
# Linear Reg Test R²: 0.8025 | MSE: 0.0346
# Decision Tree Test R²: 0.8524 | MSE: 0.0258
# Gradient Boost Test R²: 0.8641 | MSE: 0.0238 ★
